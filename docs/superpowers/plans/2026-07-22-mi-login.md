```kotlin
// Copyright 2026, compose-miuix-ui contributors
// SPDX-License-Identifier: Apache-2.0
package com.bandkit.app.core

import okhttp3.*

actual object MiClient {
    private val client = OkHttpClient.Builder()
        .followRedirects(false)
        .followSslRedirects(false)
        .build()

    actual fun get(url: String, headers: Map<String, String>, cookies: Map<String, String>): MiHttpResponse {
        val requestBuilder = Request.Builder().url(url)
        headers.forEach { (k, v) -> requestBuilder.addHeader(k, v) }
        val cookieHeader = cookies.entries.joinToString("; ") { "${it.key}=${it.value}" }
        if (cookieHeader.isNotEmpty()) requestBuilder.addHeader("Cookie", cookieHeader)
        val response = client.newCall(requestBuilder.build()).execute()
        return miResponseToData(response)
    }

    actual fun post(url: String, data: Map<String, String>, headers: Map<String, String>, cookies: Map<String, String>): MiHttpResponse {
        val formBody = FormBody.Builder().apply { data.forEach { (k, v) -> add(k, v) } }.build()
        val requestBuilder = Request.Builder().url(url).post(formBody)
        headers.forEach { (k, v) -> requestBuilder.addHeader(k, v) }
        val cookieHeader = cookies.entries.joinToString("; ") { "${it.key}=${it.value}" }
        if (cookieHeader.isNotEmpty()) requestBuilder.addHeader("Cookie", cookieHeader)
        val response = client.newCall(requestBuilder.build()).execute()
        return miResponseToData(response)
    }

    private fun miResponseToData(response: Response): MiHttpResponse {
        val cookies = response.headers("Set-Cookie").map { line ->
            val parts = line.split(";")[0].split("=", limit = 2)
            parts[0] to if (parts.size > 1) parts[1] else ""
        }.toMap()
        return MiHttpResponse(
            statusCode = response.code,
            body = response.body?.string() ?: "",
            cookies = cookies,
        )
    }
}
```

Wait - the captcha flow is different. When Xiaomi needs a captcha, the user solves the captcha in the WebView. After solving, Xiaomi redirects back to the callback URL with a serviceToken in the cookie. Let me fix:

```kotlin
@SuppressLint("SetJavaScriptEnabled")
fun showCaptchaDialog(context: Context, captchaUrl: String, callbackUrl: String, onResult: (String?) -> Unit) {
    val webView = WebView(context).apply {
        settings.javaScriptEnabled = true
    }
    val dialog = AlertDialog.Builder(context)
        .setTitle("验证码")
        .setView(webView)
        .setCancelable(false)
        .setNegativeButton("取消") { _, _ -> onResult(null) }
        .create()

    webView.webViewClient = object : WebViewClient() {
        override fun onPageStarted(view: WebView?, url: String?, favicon: android.graphics.Bitmap?) {
            super.onPageStarted(view, url, favicon)
            // 检查是否重定向到 callback URL
            if (url != null && url.startsWith(callbackUrl)) {
                // 从 cookie 中提取 serviceToken（实际在后续HTTP请求中获取）
                dialog.dismiss()
                onResult(url)  // 返回重定向后的 URL，后续从中提取 token
            }
        }
    }
    webView.loadUrl(captchaUrl)
    dialog.show()
}
```

```kotlin
private fun performLogin(email: String, password: String, onStep: (String) -> Unit, onCaptcha: (String, String) -> Unit, onSuccess: (String, String, String, String) -> Unit, onError: (String) -> Unit) {
    val deviceId = "an_" + md5(email)
    
    // Step 1: GET serviceLogin
    val step1 = MiClient.get(
        url = "https://account.xiaomi.com/pass/serviceLogin?_json=true&sid=miothealth&_locale=en_US",
        headers = mapOf("User-Agent" to "Mozilla/5.0 (Linux; Android 12; Pixel 4)..."),
        cookies = mapOf("userId" to email, "deviceId" to deviceId),
    )
    val step1Json = Json.parseToJsonElement(step1.body.substringAfter("&&&START&&&")).jsonObject
    val sign = step1Json["_sign"]?.jsonPrimitive?.content ?: throw Exception("no _sign")
    val qs = step1Json["qs"]?.jsonPrimitive?.content ?: throw Exception("no qs")
    val callback = step1Json["callback"]?.jsonPrimitive?.content ?: throw Exception("no callback")

    // Step 2: POST serviceLoginAuth2
    val md5Hash = md5(password).uppercase()
    val step2 = MiClient.post(
        url = "https://account.xiaomi.com/pass/serviceLoginAuth2",
        data = mapOf(
            "qs" to qs, "callback" to callback, "_json" to "true",
            "_sign" to sign, "user" to email, "hash" to md5Hash,
            "sid" to "miothealth", "_locale" to "en_US",
        ),
        headers = mapOf("User-Agent" to "...", "Content-Type" to "application/x-www-form-urlencoded"),
        cookies = mapOf("deviceId" to deviceId),
    )
    val step2Json = Json.parseToJsonElement(step2.body.substringAfter("&&&START&&&")).jsonObject
    val code = step2Json["code"]?.jsonPrimitive?.contentOrNull?.toIntOrNull() ?: 0
    
    if (code != 0) {
        // Need captcha
        val captchaUrl = step2Json["captcha_url"]?.jsonPrimitive?.content ?: throw Exception("captcha required but no url")
        onCaptcha(captchaUrl, callback)
        return
    }
    
    val ssec = step2Json["ssecurity"]?.jsonPrimitive?.content ?: throw Exception("no ssecurity")
    val nonce = step2Json["nonce"]?.jsonPrimitive?.contentOrNull
    val uid = step2Json["userId"]?.jsonPrimitive?.contentOrNull ?: ""
    val cUid = step2Json["cUserId"]?.jsonPrimitive?.contentOrNull ?: ""
    val location = step2Json["location"]?.jsonPrimitive?.content ?: throw Exception("no location")
}
```

登录成功后，用 `ssecurity` + `serviceToken` 调 `get_source_list`（需 RC4 加密）：

```kotlin
fun fetchDevices(ssec: String, svcToken: String, cUid: String, onDevices: (List<XiaomiDeviceInfo>) -> Unit) {
    val nonce = NativeDevice.miGenerateNonce(0)
    val params = mapOf("data" to """{"page_size":50,"status":1}""")
    val paramsJson = Json.encodeToString(params)
    val encryptedJson = NativeDevice.miEncryptParams("POST", "", paramsJson, nonce, ssec)
    val encryptedMap = Json.parseToJsonElement(encryptedJson).jsonObject
    val encryptedParams = encryptedMap.mapValues { it.value.jsonPrimitive.content }
    
    val resp = MiClient.post(
        url = "https://hlth.io.mi.com/app/v1/source/get_source_list",
        data = encryptedParams,
        cookies = mapOf("serviceToken" to svcToken, "cUserId" to cUid, "locale" to "en_us"),
        headers = mapOf("User-Agent" to "Android-12-9.8.348i-google-Pixel 4"),
    )
    
    val decrypted = NativeDevice.miDecryptResponse(resp.body, nonce, ssec)
    val json = Json.parseToJsonElement(decrypted ?: "").jsonObject
    val result = json["result"]?.jsonObject ?: return
    val list = result["list"]?.jsonArray ?: return
    
    val devices = list.mapNotNull { item ->
        val obj = item.jsonObject
        val detail = obj["detailInfo"]?.jsonObject ?: return@mapNotNull null
        XiaomiDeviceInfo(
            mac = obj["mac"]?.jsonPrimitive?.contentOrNull ?: "",
            name = obj["name"]?.jsonPrimitive?.contentOrNull ?: "",
            authKey = detail["auth_key"]?.jsonPrimitive?.contentOrNull ?: "",
        )
    }
    onDevices(devices)
}
```
